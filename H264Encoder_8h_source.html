24/**
25 * @class H264Encoder
26 * @brief Template class for capturing frames from ESP32 camera and encoding to
27 * H.264
28 *
29 * H264Encoder is a lightweight, header-only class that wraps camera
30 * initialization, pixel format conversions (RGB565/YUV422 → I420), H.264
31 * encoder lifecycle, and streaming to any Arduino Print implementation.
32 *
33 * The class is templated on an allocator type to allow customization of memory
34 * allocation strategy (e.g., PSRAM vs regular heap).
35 *
36 * @tparam Alloc Memory allocator type for internal buffers (defaults to
37 * PSRAMAllocatorH264)
38 *
39 * @note This class intentionally keeps implementation in the header for easy
40 * integration into Arduino sketches without separate .cpp files
41 * @note The pixel format conversions are straightforward and not highly
42 * optimized
43 * @note Requires ESP32 with camera module and sufficient memory for frame
44 * buffers
45 *
46 * Example usage:
47 * @code
48 * #include "H264Encoder.h"
49 * #include "UDPPrint.h"
50 *
51 * H264EncoderPSRAM streamer; // Uses PSRAM allocation (simpler than
52 * H264Encoder<>) UDPPrint udpOut;
53 *
54 * void setup() {
55 * auto cfg = streamer.defaultConfig();
56 * cfg.ssid = "MyWiFi";
57 * cfg.password = "MyPassword";
58 * cfg.width = 640;
59 * cfg.height = 480;
60 * cfg.fps = 15;
61 *
62 * udpOut.begin("192.168.1.100", 5000);
63 * if (!streamer.begin(cfg)) {
64 * Serial.println("Failed to start camera streamer");
65 * }
66 * }
67 *
68 * void loop() {
69 * streamer.captureH264(udpOut); // Captures, encodes, and streams
70 * udpOut.flush();
71 * }
72 * @endcode
73 */
74template <typename Alloc = H264_DEFAULT_ALLOCATOR>

249 /**
250 * @brief Encode raw YUV422 frame data to H.264 and write to Print stream
251 *
252 * Converts YUV422 data to I420, encodes it, and writes to the Print stream.
253 * @param raw_data Pointer to YUV422 pixel data buffer
254 * @param raw_len Size of YUV422 data in bytes
255 * @param out Print stream to receive encoded H.264 data
256 * @return true if encoding and writing succeeded, false otherwise
257 */
258 bool encodeYUV422(const uint8_t* raw_data, size_t raw_len, Print& out) {
259 ESP_LOGD(TAG, "encodeYUV422: %u bytes", (unsigned)raw_len);
260 if (!enc_handle_) {
261 ESP_LOGE(TAG, "Invalid encoder handle in encodeYUV422");
262 return false;
263 }
264 if (!raw_data || raw_len == 0) {
265 ESP_LOGE(TAG,
266 "No data provided to encodeYUV422 (raw_data=%p, raw_len=%u)",
267 raw_data, (unsigned)raw_len);
268 return false;
269 }
270 if (in_buf_.empty()) {
271 ESP_LOGE(TAG, "Input buffer is empty in encodeYUV422");
272 return false;
273 }
274 if (out_buf_.empty()) {
275 ESP_LOGE(TAG, "Output buffer is empty in encodeYUV422");
276 return false;
277 }
278 // Convert YUV422 to I420 using color conversion utility
279 // Assume width and height from config
280 size_t expected = (size_t)cfg_.width * (size_t)cfg_.height * 2;
281 if (raw_len < expected) {
282 ESP_LOGE(
283 TAG,
284 "Insufficient data for encodeYUV422: expected %u bytes, got %u bytes",
285 (unsigned)expected, (unsigned)raw_len);
286 return false;
287 }
288 // Use optimized color conversion functions from h264_color_convert.h
289#ifdef HAVE_ESP32S3
290 yuyv2iyuv_esp32s3((uint32_t)cfg_.height, (uint32_t)cfg_.width,
291 (uint8_t*)raw_data, in_buf_.data());
292#else
293 yuyv2iyuv((uint32_t)cfg_.height, (uint32_t)cfg_.width, (uint8_t*)raw_data,
294 in_buf_.data());
295#endif
296 // Now encode the I420 data
297 return encode(in_buf_.data(), in_buf_.size(), out);
298 }

353 /**
354 * @brief Encode raw I420 frame data to H.264 and write to Print stream
355 *
356 * Takes raw I420 pixel data, encodes it using the H.264 software encoder,
357 * and writes the resulting NAL units to the provided Print stream.
358 * Handles partial writes with retry logic.
359 *
360 * @param raw_data Pointer to I420 pixel data buffer
361 * @param raw_len Size of I420 data in bytes
362 * @param out Print stream to receive encoded H.264 data
363 * @return true if encoding and writing succeeded, false otherwise
364 *
365 * @note Input data must be valid I420 format with correct dimensions
366 * @note Implements retry logic for partial Print::write() operations
367 * @note This is a lower-level method; most users should use captureH264()
368 * instead
369 */
370 bool encode(const uint8_t* raw_data, size_t raw_len, Print& out) {
371 ESP_LOGD(TAG, "encode: %u bytes", (unsigned)raw_len);
372 if (!enc_handle_) {
373 ESP_LOGE(TAG, "Invalid encoder handle in encode()");
374 return false;
375 }
376 if (!raw_data || raw_len == 0) {
377 ESP_LOGE(TAG, "Invalid input data for encode(): raw_data=%p, raw_len=%u",
378 raw_data, (unsigned)raw_len);
379 return false;
380 }
381 if (out_buf_.empty()) {
382 ESP_LOGE(TAG, "Output buffer is empty in encode()");
383 return false;
384 }
385
386 esp_h264_enc_in_frame_t in_frame{};
387 esp_h264_enc_out_frame_t out_frame{};
388 in_frame.raw_data.buffer = (uint8_t*)raw_data;
389 in_frame.raw_data.len = raw_len;
390 out_frame.raw_data.buffer = out_buf_.data();
391 out_frame.raw_data.len = out_buf_.size();
392
393 ESP_LOGD(
394 TAG,
395 "Starting encoding: input size=%u bytes, output buffer size=%u bytes",
396 (unsigned)raw_len, (unsigned)out_buf_.size());
397
398 esp_h264_err_t r = esp_h264_enc_process(enc_handle_, &in_frame, &out_frame);
399 if (r != ESP_H264_ERR_OK) {
400 ESP_LOGE(TAG, "Encoding failed with error code: %d", r);
401 return false;
402 }
403
404 size_t written = 0;
405 if (out_frame.length > 0) {
406 const uint8_t* buf = out_frame.raw_data.buffer;
407 size_t to_write = out_frame.length;
408 const int max_retries = 5;
409 int attempts = 0;
410 while (written < to_write && attempts < max_retries) {
411 size_t n = out.write(buf + written, to_write - written);
412 if (n > 0) {
413 written += n;
414 } else {
415 // if nothing was written, wait a bit before retrying to avoid busy
416 // loop
417 delay(5);
418 }
419 ++attempts;
420 }
421 if (written < to_write) {
422 ESP_LOGW(TAG, "Partial write: expected %u bytes, wrote %u bytes",
423 (unsigned)to_write, (unsigned)written);
424 return false;
425 }
426 }
427 ESP_LOGI(TAG, "Encoded frame: i420=%u bytes, h264=%u bytes",
428 (unsigned)in_frame.raw_data.len, (unsigned)out_frame.length);
429
430 return (r == ESP_H264_ERR_OK);
431 }

433 /**
434 * @brief Capture frame, encode to H.264, and stream with frame rate control
435 *
436 * High-level method that performs the complete streaming pipeline:
437 * 1. Captures one frame from the camera
438 * 2. Converts pixel format to I420 if needed
439 * 3. Encodes frame using H.264 software encoder
440 * 4. Writes encoded data to the provided Print stream
441 * 5. Enforces configured frame rate by delaying remaining time
442 *
443 * This method measures processing time and only delays for the remainder
444 * of the target frame interval, ensuring consistent frame rate regardless
445 * of processing time variations.
446 *
447 * @param out Print stream to receive encoded H.264 NAL units
448 * @return true if capture, encoding, and streaming succeeded, false otherwise
449 *
450 * @note Frame rate is controlled by Config::fps setting
451 * @note Processing time is subtracted from frame interval delay
452 * @note This is the main method most applications should use
453 * @note On failure, no delay is applied (to avoid blocking on errors)
454 */
455 bool captureH264(Print& out) {
456 ESP_LOGD(TAG, "captureH264");
457 // basic guards
458 if (in_buf_.empty()) return false;
459 if (!enc_handle_) return false;
460
461 // compute target frame interval in ms
462 unsigned long frame_ms = (cfg_.fps > 0) ? (1000u / (unsigned)cfg_.fps) : 0u;
463 unsigned long start = millis();
464
465 // capture into the input buffer
466 if (!captureFrameI420(in_buf_.data(), in_buf_.size())) {
467 return false;
468 }
469
470 // encode from the input buffer to the output buffer, writing to `out`
471 if (!encode(in_buf_.data(), in_buf_.size(), out)) {
472 return false;
473 }
474
475 // enforce configured frame rate: delay only remaining time after processing
476 if (frame_ms > 0) {
477 unsigned long elapsed = millis() - start;
478 if (elapsed < frame_ms) delay(frame_ms - elapsed);
479 }
480
481 return true;
482 }

567 /**
568 * @brief Initialize ESP32 camera with Config pin mappings and settings
569 * @return true if camera initialization successful and buffers allocated,
570 * false on error
571 * @note Allocates I420 conversion buffer (width×height×1.5 bytes) and output
572 * buffer using configured allocator
573 * @note Automatically selects framesize based on Config::width and
574 * Config::height
575 * @note Sets RGB565 pixel format for color conversion compatibility
576 */
577 bool initCamera() {
578 ESP_LOGD(TAG, "initCamera");
579 // choose framesize from requested width/height; default to VGA
580 auto pick_frame_size = [](int w, int h) -> framesize_t {
581 if (w == 160 && h == 120) return FRAMESIZE_QQVGA;
582 if (w == 320 && h == 240) return FRAMESIZE_QVGA;
583 if (w == 640 && h == 480) return FRAMESIZE_VGA;
584 if (w == 800 && h == 600) return FRAMESIZE_SVGA;
585 if (w == 1024 && h == 768) return FRAMESIZE_XGA;
586 // fallback
587 ESP_LOGI(TAG, "Unsupported resolution %dx%d, defaulting to VGA", w, h);
588 return FRAMESIZE_VGA;
589 };
590 framesize_t fs = pick_frame_size(cfg_.width, cfg_.height);
591 if (!tryInitCamera(fs, cfg_, "current-config")) {
592 for (const auto& mapping : kFallbackMappings) {
593 Config fallback = cfg_;
594 applyPinMapping(fallback, mapping);
595 if (samePins(cfg_, fallback)) {
596 continue;
597 }
598 if (tryInitCamera(fs, fallback, mapping.name)) {
599 cfg_ = fallback;
600 break;
601 }
602 }
603
604 if (!cameraInitialized()) {
605 ESP_LOGE(TAG,
606 "esp_camera_init failed for all known ESP32-S3 camera pin "
607 "mappings");
608 ESP_LOGE(
609 TAG,
610 "Likely causes: unsupported camera sensor in esp_camera, "
611 "custom/non-standard ESP32-S3 board pinout, or incorrect wiring");
612 return false;
613 }
614 }
615 ESP_LOGD(TAG, "Camera initialized successfully");
616
617 sensor = esp_camera_sensor_get();
618 if (sensor) {
619 ESP_LOGI(TAG, "PID: 0x%02X", sensor->id.PID);
620 sensor->set_framesize(sensor, fs);
621 }
622
623 return true;
624 }

746 /**
747 * @brief Initialize H.264 software encoder with Config settings
748 * @return true if encoder creation and opening successful, false on error
749 * @note Automatically calculates bitrate as width×height×fps÷20
750 * @note Sets GOP (Group of Pictures) to 2×fps for balanced
751 * compression/quality
752 * @note Uses esp_h264 software encoder (not hardware acceleration)

755 ESP_LOGD(TAG, "initEncoder");
756
757 esp_h264_enc_cfg_sw_t enc_cfg{};
758 enc_cfg.res.width = (uint16_t)cfg_.width;
759 enc_cfg.res.height = (uint16_t)cfg_.height;
760 enc_cfg.fps = (uint8_t)cfg_.fps;
761 enc_cfg.gop = (cfg_.gop > 0) ? (uint8_t)cfg_.gop : (uint8_t)cfg_.fps;
762 enc_cfg.pic_type = ESP_H264_RAW_FMT_I420;
763 enc_cfg.rc.bitrate =
764 (cfg_.bitrate > 0)
765 ? (uint32_t)cfg_.bitrate
766 : (uint32_t)(cfg_.width * cfg_.height * cfg_.fps * 30 / 100);
767 enc_cfg.rc.qp_min = (cfg_.qp_min >= 0) ? (uint8_t)cfg_.qp_min : 28;
768 enc_cfg.rc.qp_max = (cfg_.qp_max >= 0) ? (uint8_t)cfg_.qp_max : 30;
769
770 ESP_LOGI(TAG,
771 "Encoder settings: width=%u, height=%u, fps=%u, gop=%u, "
772 "pic_type=0x%08X, bitrate=%u, qp_min=%u, qp_max=%u",
773 (unsigned)enc_cfg.res.width, (unsigned)enc_cfg.res.height,
774 (unsigned)enc_cfg.fps, (unsigned)enc_cfg.gop,
775 (unsigned)enc_cfg.pic_type, (unsigned)enc_cfg.rc.bitrate,
776 (unsigned)enc_cfg.rc.qp_min, (unsigned)enc_cfg.rc.qp_max);
777
778 esp_h264_err_t res = esp_h264_enc_sw_new(&enc_cfg, &enc_handle_);
779 if (res != ESP_H264_ERR_OK || !enc_handle_) {
780 ESP_LOGE(TAG, "esp_h264_enc_sw_new failed: %d", (int)res);

786 return false;
787 }
788 return true;

791 /**
792 * @brief Capture one frame and convert to I420 (YUV420 planar) format
793 *
794 * Captures frame from ESP32 camera and converts from the camera's native
795 * pixel format to I420 format required by H.264 encoder. Supports RGB565
796 * and YUV422 input formats with automatic format detection.
797 *
798 * @param dst Destination buffer for I420 data (must be width×height×1.5
799 * bytes)
800 * @param dst_len Size of destination buffer in bytes
801 * @return true if frame captured and converted successfully, false on error
802 *
803 * @note I420 format: Y plane (width×height), U plane (width/2×height/2), V
804 * plane (width/2×height/2)
805 * @note Automatically returns camera framebuffer after conversion
806 * @note Logs error for unsupported pixel formats
807 */
808 bool captureFrameI420(uint8_t* dst, size_t dst_len) {
809 ESP_LOGD(TAG, "captureFrameI420");
810 if (!dst) return false;
811 camera_fb_t* fb = esp_camera_fb_get();
812 if (!fb) return false;
813
814 int w = fb->width;
815 int h = fb->height;
816 size_t expected = (size_t)w * (size_t)h * 3 / 2;
817 if (dst_len < expected) {
818 ESP_LOGE(TAG, "Destination buffer too small: need %u got %u",
819 (unsigned)expected, (unsigned)dst_len);
820 esp_camera_fb_return(fb);
821 return false;
822 }
823
824 bool res = false;
825 switch (fb->format) {
826 case PIXFORMAT_RGB565:
827 res = captureFrameI420_RGB565(dst, dst_len, fb, w, h);
828 break;
829 case PIXFORMAT_YUV422:
830 res = captureFrameI420_YUV422(dst, dst_len, fb, w, h);
831 break;
832 default:
833 ESP_LOGE(TAG, "Unsupported camera pixel format: %d", fb->format);
834 res = false;
835 esp_camera_fb_return(fb);
836 break;
837 }
838
839 return res;
840 }

842 /**
843 * @brief Convert raw RGB565 buffer to I420 (YUV420 planar) format
844 * @param src Pointer to RGB565 input buffer
845 * @param dst Pointer to I420 output buffer
846 * @param w Frame width in pixels
847 * @param h Frame height in pixels
848 */
849 void rgb565_to_i420(const uint8_t* src, uint8_t* dst, int w, int h) {
850 // --- Y plane ---
851 for (int y = 0; y < h; ++y) {
852 for (int x = 0; x < w; ++x) {
853 int i = (y * w + x) * 2;
854 // little endien
855 uint16_t val = src[i] | (src[i + 1] << 8);
856 // big endian
857 //uint16_t val = (src[i] << 8) | src[i + 1];
858
859 // Better RGB expansion
860 uint8_t r = ((val >> 11) & 0x1F);
861 uint8_t g = ((val >> 5) & 0x3F);
862 uint8_t b = (val & 0x1F);
863
864 r = (r << 3) | (r >> 2);
865 g = (g << 2) | (g >> 4);
866 b = (b << 3) | (b >> 2);
867
868 int yv = ((66 * r + 129 * g + 25 * b + 128) >> 8) + 16;
869 dst[y * w + x] = (uint8_t)std::clamp(yv, 0, 255);
870 }
871 }
872
873 uint8_t* uplane = dst + w * h;
874 uint8_t* vplane = uplane + (w / 2) * (h / 2);
875
876 // uint8_t* vplane = dst + w * h;
877 // uint8_t* uplane = vplane + (w / 2) * (h / 2);
878
879 // --- UV planes (4:2:0 subsampling) ---
880 for (int y = 0; y < h; y += 2) {
881 for (int x = 0; x < w; x += 2) {
882 int sumU = 0, sumV = 0;
883
884 for (int dy = 0; dy < 2; ++dy) {
885 for (int dx = 0; dx < 2; ++dx) {
886 int i = ((y + dy) * w + (x + dx)) * 2;
887 uint16_t val = src[i] | (src[i + 1] << 8);
888
889 uint8_t r = ((val >> 11) & 0x1F);
890 uint8_t g = ((val >> 5) & 0x3F);
891 uint8_t b = (val & 0x1F);
892
893 r = (r << 3) | (r >> 2);
894 g = (g << 2) | (g >> 4);
895 b = (b << 3) | (b >> 2);
896
897 int u = ((-38 * r - 74 * g + 112 * b + 128) >> 8) + 128;
898 int v = ((112 * r - 94 * g - 18 * b + 128) >> 8) + 128;
899
900 sumU += u;
901 sumV += v;
902 }
903 }
904
905 int ux = x / 2;
906 int uy = y / 2;
907 int pos = uy * (w / 2) + ux;
908
909 // Clamp AFTER averaging
910 uplane[pos] = (uint8_t)std::clamp(sumU / 4, 0, 255);
911 vplane[pos] = (uint8_t)std::clamp(sumV / 4, 0, 255);
912 }
913 }
914 }
