20/**
21 * @class RAMAllocatorH264
22 * @brief STL-compatible allocator that uses internal RAM (DRAM) allocation
23 *
24 * RAMAllocatorH264 is a custom C++ allocator that allocates memory from internal
25 * DRAM using heap_caps_aligned_calloc with MALLOC_CAP_INTERNAL. This ensures
26 * allocation from fast internal memory rather than external PSRAM.
27 *
28 * This allocator is useful for ESP32 platforms when you specifically want
29 * small, frequently-accessed buffers to be stored in fast internal RAM
30 * rather than slower external PSRAM.
31 *
32 * @tparam T The type of objects to allocate
33 *
34 * @note On ESP-IDF platforms, uses heap_caps_aligned_calloc for internal RAM
35 * allocation
36 * @note On Arduino platforms, the heap_caps functions are shimmed to standard
37 * malloc/free
38 * @note Memory is zero-initialized (calloc behavior)
39 * @note Provides proper alignment for the allocated type
40 *
41 * Example usage:
42 * @code
43 * #include "RAMAllocatorH264.h"
44 *
45 * // Use with std::vector for small, fast buffers
46 * std::vector<uint32_t, RAMAllocatorH264<uint32_t>> smallBuffer;
47 * smallBuffer.resize(256); // Small buffer in fast internal RAM
48 *
49 * // Or use the convenience alias
50 * RAMVec<uint32_t> fastBuffer;
51 * fastBuffer.resize(256);
52 * @endcode
53 */

56 public:

79 /**
80 * @brief Allocate memory for n objects of type T in internal RAM
81 *
82 * Attempts to allocate memory in the following priority order:
83 * 1. Internal RAM using heap_caps_aligned_calloc (if available)
84 * 2. Aligned allocation using aligned_alloc (C11)
85 * 3. Standard calloc as fallback
86 *
87 * All allocated memory is zero-initialized.
88 *
89 * @param n Number of objects of type T to allocate
90 * @return Pointer to allocated memory
91 * @throws std::bad_alloc if allocation fails
92 *
93 * @note Memory is aligned by 16
94 * @note On ESP32, prefers internal DRAM allocation over PSRAM
95 */
96 T* allocate(std::size_t n) {
97 if (n == 0) return nullptr;
98 uint32_t actual_size = 0;
99 void* p = esp_h264_aligned_calloc(
100 16, 1, n * sizeof(T), &actual_size, (uint32_t)MALLOC_CAP_INTERNAL);
101 return static_cast<T*>(p);
102 }
103
104 /**
105 * @brief Deallocate memory previously allocated by this allocator
106 *
107 * Frees memory that was allocated by allocate(). Uses heap_caps_free
108 * which works for both internal RAM and regular heap allocations.
109 *
110 * @param p Pointer to memory to deallocate (can be nullptr)
111 * @param n Number of objects (ignored, but required by allocator interface)
112 *
113 * @note Safe to call with nullptr
114 * @note The size parameter n is ignored as heap_caps_free doesn't need it
115 */
116 void deallocate(T* p, std::size_t /*n*/) noexcept {
117 if (!p) return;
118 esp_h264_free_internal(p);
119 }
