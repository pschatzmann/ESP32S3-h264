8 * @brief RDPPacketizer class to packetize H.264 NAL units into RTP packets.
9 * This class takes raw H.264 NAL units (in Annex-B format) and packetizes them
10 * into RTP packets suitable for streaming over UDP.
11 *
12 * This way you can e.g. Send H264 frames to QGroundcontrol
13 */

26 // Call once per encoded frame
27 size_t write(const uint8_t* data, size_t len) {
28 if (!data || len < 5) return 0;
29 ESP_LOGI(TAG, "Packetizing frame of size %u bytes", (unsigned)len);
30
31 // Stable timestamp (better than millis jitter)
32 timestamp_ += 3000; // ~30 FPS (90000 / 30)
33
34 // Parse Annex-B and extract ALL NAL units in the frame
35 std::vector<std::pair<const uint8_t*, size_t>> nals;
36 extractNALs(data, len, nals);
37 if (nals.empty()) return 0;
38
39 // Detect IDR
40 bool is_idr = false;
41 for (auto& n : nals) {
42 uint8_t type = n.first[0] & 0x1F;
43 if (type == 5) is_idr = true;
44 if (type == 7) sps_.assign(n.first, n.first + n.second);
45 if (type == 8) pps_.assign(n.first, n.first + n.second);
46 }
47
48 // Inject SPS/PPS before IDR
49 std::vector<std::pair<const uint8_t*, size_t>> send_list;
50
51 if (is_idr && !sps_.empty() && !pps_.empty()) {
52 send_list.emplace_back(sps_.data(), sps_.size());
53 send_list.emplace_back(pps_.data(), pps_.size());
54 }
55
56 for (auto& n : nals) {
57 send_list.push_back(n);
58 }
59
60 // Send all NALs, only LAST packet gets marker bit
61 for (size_t i = 0; i < send_list.size(); ++i) {
62 bool last_nal = (i == send_list.size() - 1);
63 sendNAL(send_list[i].first, send_list[i].second, last_nal);
64 }
65 return len;
66 }

91 void sendPacket(const uint8_t* payload, size_t payload_len, bool marker) {
92 uint8_t rtp[12];
93 buildHeader(rtp, marker);
94 buffer_.clear();
95 buffer_.insert(buffer_.end(), rtp, rtp + 12);
96 buffer_.insert(buffer_.end(), payload, payload + payload_len);
97 out_.write(buffer_.data(), buffer_.size());
98
99 sequence_number_++;
100 }

102 void sendNAL(const uint8_t* data, size_t len, bool last_nal_of_frame) {
103 if (len <= MTU - 12) {
104 sendPacket(data, len, last_nal_of_frame);
105 return;
106 }
107
108 // FU-A fragmentation
109 uint8_t nal_header = data[0];
110 uint8_t nal_type = nal_header & 0x1F;
111
112 size_t payload_max = MTU - 12 - 2;
113 size_t pos = 1;
114 bool start = true;
115
116 while (pos < len) {
117 size_t frag_len = std::min(payload_max, len - pos);
118 bool end = (pos + frag_len) >= len;
119
120 uint8_t fua[2];
121 fua[0] = (nal_header & 0xE0) | 28;
122 fua[1] = (start ? 0x80 : 0x00) | (end ? 0x40 : 0x00) | nal_type;
123
124 uint8_t buffer[MTU];
125 memcpy(buffer, fua, 2);
126 memcpy(buffer + 2, data + pos, frag_len);
127
128 bool marker = last_nal_of_frame && end;
129 sendPacket(buffer, frag_len + 2, marker);
130
131 pos += frag_len;
132 start = false;
133 }
134 }

136 void buildHeader(uint8_t* hdr, bool marker) {
137 hdr[0] = (RTP_VERSION << 6);
138 hdr[1] = (marker ? 0x80 : 0x00) | (RTP_PAYLOAD_TYPE & 0x7F);
139
140 hdr[2] = sequence_number_ >> 8;
141 hdr[3] = sequence_number_ & 0xFF;
142
143 hdr[4] = (timestamp_ >> 24) & 0xFF;
144 hdr[5] = (timestamp_ >> 16) & 0xFF;
145 hdr[6] = (timestamp_ >> 8) & 0xFF;
146 hdr[7] = (timestamp_) & 0xFF;
147
148 hdr[8] = (ssrc_ >> 24) & 0xFF;
149 hdr[9] = (ssrc_ >> 16) & 0xFF;
150 hdr[10] = (ssrc_ >> 8) & 0xFF;
151 hdr[11] = (ssrc_) & 0xFF;
152 }

156 // -------------------------

158 void extractNALs(const uint8_t* data, size_t len,
159 std::vector<std::pair<const uint8_t*, size_t>>& out) {
160 size_t i = 0;
161
162 while (i + 4 < len) {
163 // find start code
164 size_t start = i;
165 while (start + 3 < len &&
166 !(data[start] == 0 && data[start + 1] == 0 &&
167 (data[start + 2] == 1 ||
168 (data[start + 2] == 0 && data[start + 3] == 1)))) {
169 start++;
170 }
171
172 if (start + 3 >= len) break;
173
174 size_t sc_size = (data[start + 2] == 1) ? 3 : 4;
175 size_t nal_start = start + sc_size;
176
177 size_t next = nal_start;
178 while (next + 3 < len &&
179 !(data[next] == 0 && data[next + 1] == 0 &&
180 (data[next + 2] == 1 ||
181 (data[next + 2] == 0 && data[next + 3] == 1)))) {
182 next++;
183 }
184
185 size_t nal_len = next - nal_start;
186 if (nal_len > 0) {
187 out.emplace_back(data + nal_start, nal_len);
188 }
189
190 i = next;
191 }
192 }
