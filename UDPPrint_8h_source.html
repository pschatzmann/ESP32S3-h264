14/**
15 * @brief A Print implementation that buffers and sends data over UDP
16 *
17 * UDPPrint extends Arduino's Print class to send data over UDP networks.
18 * It internally buffers data and sends UDP packets when the buffer fills
19 * or when explicitly flushed.
20 *
21 * @note WiFi connection must be established before using this class
22 * @note Data is buffered up to BUF_SIZE (1400 bytes) before automatic
23 * transmission
24 *
25 * Example usage:
26 * @code
27 * #include "UDPPrint.h"
28 *
29 * UDPPrint up;
30 * up.begin("192.168.1.100", 5000);
31 * up.print("Hello World");
32 * up.flush(); // Send immediately
33 * @endcode
34 */

75 // Write a single byte
76 size_t write(uint8_t ch) override {
77 buffer_[bufPos_++] = ch;
78 if (bufPos_ >= sizeof(buffer_)) flush();
79 return 1;
80 }
81
82 // Write a buffer
83 size_t write(const uint8_t* data, size_t size) override {
84 size_t written = 0;
85 while (size > 0) {
86 size_t space = sizeof(buffer_) - bufPos_;
87 if (space == 0) flush();
88 size_t tocopy = (size < space) ? size : space;
89 memcpy(buffer_ + bufPos_, data + written, tocopy);
90 bufPos_ += tocopy;
91 written += tocopy;
92 size -= tocopy;
93 }
94 flush();
95 ESP_LOGI("UDPPrint", "Buffered and sent %u bytes to %s:%u",
96 (unsigned)written, destIp_.c_str(), destPort_);
97 return written;
98 }
99
100 // Send buffered data immediately
101 void flush() {
102 if (bufPos_ == 0) return;
103 if (!udp_) return;
104
105 size_t to_send = bufPos_;
106
107 bool sent = false;
108 size_t written = 0;
109 int retry = 10;
110 while (!sent && retry-- > 0) {
111 if (!udp_->beginPacket(destIp_.c_str(), destPort_)) {
112 // unable to start packet
113 bufPos_ = 0;
114 return;
115 }
116 written = udp_->write(buffer_, bufPos_);
117 sent = udp_->endPacket();
118 delay(write_delay_ms);
119 }
120
121 if (!sent) {
122 ESP_LOGE("UDPPrint",
123 "Failed to send UDP packet %u bytes to %s:%u after retries",
124 (unsigned)to_send, destIp_.c_str(), destPort_);
125 }
126 bufPos_ = 0;
127 }
128
129 // Convenience: send immediate string
130 size_t print(const char* s) {
131 size_t len = write((const uint8_t*)s, strlen(s));
132 flush();
133 return len;
134 }
135
136 size_t println(const char* s) {
137 char buffer[strlen(s) + 2]{};
138 strcpy(buffer, s);
139 strcat(buffer, "\n");

141 }
